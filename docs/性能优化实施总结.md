def get_statistics_by_task(self, task_id: int) -> dict:
    from sqlalchemy import func, case

    result = self.db.query(
        func.count(WebScraperLog.id).label('total'),
        func.sum(case((WebScraperLog.status == LogStatus.SUCCESS, 1), else_=0)).label('success'),
        func.sum(case((WebScraperLog.status == LogStatus.FAILED, 1), else_=0)).label('failed'),
        func.sum(case((WebScraperLog.status == LogStatus.RUNNING, 1), else_=0)).label('running')
    ).filter(WebScraperLog.task_id == task_id).first()
    # 单次查询完成所有统计

#### 1.3 Repository查询优化
**文件**:
- `backend/app/repositories/web_scraper_task_repository.py:30-40`
- `backend/app/repositories/web_scraper_log_repository.py:30-36`

**修改内容**:
添加了可选的预加载参数，避免N+1查询问题：

```python
# web_scraper_task_repository.py
def get_by_id(self, task_id: int, with_relations: bool = False) -> Optional[WebScraperTask]:
    query = self.db.query(WebScraperTask)

    if with_relations:
        query = query.options(
            joinedload(WebScraperTask.knowledge_base),
            joinedload(WebScraperTask.creator)
        )

    return query.filter(WebScraperTask.id == task_id).first()

# web_scraper_log_repository.py
def get_by_id(self, log_id: int, with_task: bool = False) -> Optional[WebScraperLog]:
    query = self.db.query(WebScraperLog)

    if with_task:
        query = query.options(joinedload(WebScraperLog.task))

    return query.filter(WebScraperLog.id == log_id).first()
```

**性能提升**:
- 避免N+1查询问题
- 减少数据库往返次数
- 预计减少30-50%的查询时间（当需要关联数据时）
